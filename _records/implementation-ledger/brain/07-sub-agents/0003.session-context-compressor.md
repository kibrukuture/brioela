# 0003 — SessionContextCompressor (Ephemeral Sub-Agent)

## Purpose

Implement the `SessionContextCompressor` — an ephemeral sub-agent invoked inline by the session lifecycle when the turn count crosses the compression threshold (defined in spec: check `17-session-lifecycle.md` for exact number).

This is not a Durable Object. It is a one-shot Haiku call that receives the current session turns and returns a 4-field compression record:

1. `topics` — main subjects discussed
2. `decisions` — things the user decided or committed to
3. `open_items` — unresolved threads
4. `behavior_signals` — observations about user patterns worth remembering

The output is written to `sessions.outcomeSummary` (JSON blob). The session is then marked `wasCompressed = true`. A new child session is opened carrying the summary as its opening context, and the original session is closed.

This sub-agent uses `claude-haiku-4-5` specifically — not the main Brain model. Haiku is cheaper and fast enough for summarization. The call is made with `maxTokens: 512` — summaries must be tight.

```typescript
import { generateText } from 'ai'
import { createAnthropic } from '@ai-sdk/anthropic'
import { z } from '@brioela/shared/zod'
import type { BrainDatabase } from '@/agents/brain/_database'
import { readSessionTurns, updateSessionCompressed, insertSession } from '@/agents/brain/_repositories'
import { createId } from '@brioela/shared/_ids'
import { readCurrentEpochMs } from '@/time/_helpers'

const compressionOutputSchema = z.object({
	topics: z.array(z.string()),
	decisions: z.array(z.string()),
	open_items: z.array(z.string()),
	behavior_signals: z.array(z.string()),
})

export async function compressSession(
	db: BrainDatabase,
	sessionId: string,
	userId: string,
	anthropicApiKey: string,
): Promise<{ newSessionId: string; summary: z.infer<typeof compressionOutputSchema> }> {
	const turns = readSessionTurns(db, sessionId)
	const transcript = turns
		.map(t => `[${t.role.toUpperCase()}] ${t.content}`)
		.join('\n\n')

	const anthropic = createAnthropic({ apiKey: anthropicApiKey })

	const { text } = await generateText({
		model: anthropic('claude-haiku-4-5-20251001'),
		maxTokens: 512,
		messages: [
			{
				role: 'user',
				content: `Compress this session into a structured JSON summary with exactly four fields: topics (array of strings), decisions (array of strings), open_items (array of strings), behavior_signals (array of strings). Be concise — each item is one sentence max. Return only valid JSON, no markdown fences.\n\n${transcript}`,
			},
		],
	})

	const summary = compressionOutputSchema.parse(JSON.parse(text))
	const now = readCurrentEpochMs()

	updateSessionCompressed(db, sessionId, {
		outcomeSummary: JSON.stringify(summary),
		wasCompressed: true,
		status: 'completed',
		endedAt: now,
		updatedAt: now,
	})

	const newSessionId = createId()
	insertSession(db, {
		id: newSessionId,
		userId,
		sessionType: 'general',
		status: 'active',
		parentSessionId: sessionId,
		outcomeSummary: null,
		turnCount: 0,
		wasCompressed: false,
		startedAt: now,
		endedAt: null,
		updatedAt: now,
	})

	return { newSessionId, summary }
}
```
