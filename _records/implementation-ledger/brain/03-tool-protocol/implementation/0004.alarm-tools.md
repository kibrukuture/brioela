# 0004 — Alarm Tools

## Purpose

Implement the two AI-callable alarm tools: `schedule_user_alarm` and `cancel_user_alarm`.

Alarms are future-trigger actions implemented via Cloudflare Durable Object alarms (`this.ctx.storage.setAlarm()`). The DB row records intent; the DO alarm fires it. The `alarm_type` field routes dispatch: `session_watchdog`, `user_reminder`, `brain_maintenance`, `behavior_pattern_detection`, and `cooking_reminder` are the known types.

The `session_type` produced when an alarm fires is `'alarm'` — it has its own session kind and its own tool set (read-only, no skill write, no constraint write).

`cancel_user_alarm` sets `status = 'cancelled'` in DB and calls `this.ctx.storage.deleteAlarm()` if the matching DO alarm is still pending. Only `user_reminder` and `cooking_reminder` alarms can be cancelled by the agent — internal system alarms (`session_watchdog`, `brain_maintenance`, `behavior_pattern_detection`) are not cancellable via this tool.

```typescript
import { and, eq, gt } from '@brioela/shared/drizzle'
import { alarms } from '@/agents/brain/_schemas'
import type { BrainDatabase } from '@/agents/brain/_database'

export function readUserAlarm(db: BrainDatabase, id: string) {
	return db.select().from(alarms).where(eq(alarms.id, id)).get()
}

export function listPendingUserAlarms(db: BrainDatabase, userId: string, afterMs: number) {
	return db.select()
		.from(alarms)
		.where(and(eq(alarms.userId, userId), eq(alarms.status, 'pending'), gt(alarms.fireAt, afterMs)))
		.all()
}
```

### 3. [NEW] [schedule.user.alarm.tool.ts](file:///Users/kibrukuture/Desktop/brioela/backend/src/agents/brain/_tools/schedule.user.alarm.tool.ts)

```typescript
import { tool } from 'ai'
import { z } from '@brioela/shared/zod'
import { createId } from '@brioela/shared/_ids'
import type { BrainDatabase } from '@/agents/brain/_database'
import { insertUserAlarm } from '@/agents/brain/_repositories'
import { readCurrentEpochMs } from '@/time/_helpers'

const scheduleUserAlarmSchema = z.object({
	label: z.string().min(1).max(120).describe('Human-readable label — shown to user when alarm fires.'),
	fire_at_ms: z.number().int().positive().describe('Epoch ms timestamp when alarm should fire. Must be in the future.'),
	alarm_type: z.enum(['user_reminder', 'cooking_reminder']).describe('Agent can only schedule user-facing alarms — not internal system alarms.'),
	payload: z.record(z.string(), z.unknown()).optional().describe('Optional structured data passed to the alarm handler.'),
	reason: z.string().min(1).describe('Why this alarm is being set.'),
})

export const scheduleUserAlarmTool = (
	db: BrainDatabase,
	userId: string,
	scheduleDoAlarm: (fireAtMs: number, alarmId: string) => Promise<void>,
) => tool({
	description: 'Schedule a future alarm for the user. The DO alarm fires and dispatches via alarm.handler.ts. Only user-facing alarm types are allowed here.',
	inputSchema: scheduleUserAlarmSchema,
	execute: async (input) => {
		const now = readCurrentEpochMs()
		if (input.fire_at_ms <= now) {
			return { error: 'fire_at_must_be_future', fire_at_ms: input.fire_at_ms, now }
		}
		const alarm = insertUserAlarm(db, {
			id: createId(),
			userId,
			label: input.label,
			alarmType: input.alarm_type,
			status: 'pending',
			fireAt: input.fire_at_ms,
			payload: input.payload ? JSON.stringify(input.payload) : null,
			reason: input.reason,
			firedAt: null,
			cancelledAt: null,
			createdAt: now,
			updatedAt: now,
		})
		await scheduleDoAlarm(input.fire_at_ms, alarm.id)
		return {
			id: alarm.id,
			status: 'pending',
			label: alarm.label,
			fire_at_ms: alarm.fireAt,
			alarm_type: alarm.alarmType,
		}
	},
})
```

### 4. [NEW] [cancel.user.alarm.tool.ts](file:///Users/kibrukuture/Desktop/brioela/backend/src/agents/brain/_tools/cancel.user.alarm.tool.ts)

```typescript
import { tool } from 'ai'
import { z } from '@brioela/shared/zod'
import type { BrainDatabase } from '@/agents/brain/_database'
import { readUserAlarm, updateUserAlarmStatus } from '@/agents/brain/_repositories'
import { readCurrentEpochMs } from '@/time/_helpers'

const cancelUserAlarmSchema = z.object({
	id: z.string().describe('Alarm id from schedule_user_alarm response.'),
	reason: z.string().min(1).describe('Why this alarm is being cancelled.'),
})

const CANCELLABLE_ALARM_TYPES = new Set(['user_reminder', 'cooking_reminder'])

export const cancelUserAlarmTool = (
	db: BrainDatabase,
	cancelDoAlarm: (alarmId: string) => Promise<void>,
) => tool({
	description: 'Cancel a pending user alarm. Only user_reminder and cooking_reminder types are cancellable. Internal system alarms cannot be cancelled via this tool.',
	inputSchema: cancelUserAlarmSchema,
	execute: async (input) => {
		const alarm = readUserAlarm(db, input.id)
		if (!alarm) return { error: 'alarm_not_found', id: input.id }
		if (alarm.status !== 'pending') return { error: 'alarm_not_pending', id: input.id, current_status: alarm.status }
		if (!CANCELLABLE_ALARM_TYPES.has(alarm.alarmType)) {
			return { error: 'alarm_type_not_cancellable', alarm_type: alarm.alarmType, hint: 'Only user_reminder and cooking_reminder can be cancelled.' }
		}
		const now = readCurrentEpochMs()
		updateUserAlarmStatus(db, input.id, { status: 'cancelled', cancelledAt: now, updatedAt: now })
		await cancelDoAlarm(input.id)
		return { id: input.id, status: 'cancelled', label: alarm.label, reason: input.reason }
	},
})
```

### 5. [MODIFY] [memory.tool.ts](file:///Users/kibrukuture/Desktop/brioela/backend/src/agents/brain/_tools/memory.tool.ts)

Add alarm tools to `buildToolsForSession`. `general` session gets both. `cooking` gets `schedule_user_alarm` (cooking_reminder type) and `cancel_user_alarm`. `alarm` session gets neither (no tool calls from alarm sessions). `brain_maintenance` gets neither.

Note: `scheduleDoAlarm` and `cancelDoAlarm` callbacks must be injected from the Durable Object layer — they call `this.ctx.storage.setAlarm()` / `this.ctx.storage.deleteAlarm()`. The tool layer never touches DO storage directly.
