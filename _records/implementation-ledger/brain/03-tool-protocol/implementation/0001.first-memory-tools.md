# 0001 - First Memory Tools

## Purpose

Implement the first slice of internal AI-callable tools for the `BrioelaBrain` Durable Object (`write_user_memory`, `read_user_memory`, and `log_memory_event`), keeping LLM read/write interactions with SQLite strictly validated, typed, and permission-controlled. 

To satisfy the "Hard Bans" against importing Drizzle schema tables directly in business features, we will also create the corresponding repositories for accessing `user_memory` records.

### 1. [NEW] [read.brain.user.memory.repository.ts](file:///Users/kibrukuture/Desktop/brioela/backend/src/agents/brain/_repositories/read.brain.user.memory.repository.ts)
Provides standard read-only access to `user_memory` rows.

```typescript
import { and, eq, getOne } from '@/database/drizzle/_database'
import { sql } from '@/database/sqlite/_schema'
import { userMemory } from '@/agents/brain/_schema'
import type { BrainDatabase } from '@/agents/brain/_database'

export function readBrainUserMemory(
	database: BrainDatabase,
	userId: string,
	namespace: string,
	key: string,
) {
	return getOne(
		database
			.select()
			.from(userMemory)
			.where(
				and(
					eq(userMemory.id, `${namespace}:${key}`),
					eq(userMemory.userId, userId),
					eq(userMemory.isActive, true),
				),
			),
	)
}

export function listBrainUserMemories(
	database: BrainDatabase,
	userId: string,
	namespace: string,
) {
	return database
		.select()
		.from(userMemory)
		.where(
			and(
				eq(userMemory.userId, userId),
				eq(userMemory.namespace, namespace),
				eq(userMemory.isActive, true),
			),
		)
		.all()
}

export function countBrainUserMemoryNamespaces(
	database: BrainDatabase,
	userId: string,
): number {
	const countRecord = getOne(
		database
			.select({
				count: sql<number>`count(distinct ${userMemory.namespace})`,
			})
			.from(userMemory)
			.where(
				and(
					eq(userMemory.userId, userId),
					eq(userMemory.isActive, true),
				),
			),
	)
	return countRecord?.count ?? 0
}
```

### 2. [NEW] [write.brain.user.memory.repository.ts](file:///Users/kibrukuture/Desktop/brioela/backend/src/agents/brain/_repositories/write.brain.user.memory.repository.ts)
Handles insertion, updating, and fire-and-forget stat updates of `user_memory`.

```typescript
import { eq, getReturned } from '@/database/drizzle/_database'
import { sql } from '@/database/sqlite/_schema'
import { userMemory, type BrainUserMemory, type NewBrainUserMemory } from '@/agents/brain/_schema'
import type { BrainDatabase } from '@/agents/brain/_database'

export function writeBrainUserMemory(
	database: BrainDatabase,
	newMemory: NewBrainUserMemory,
): BrainUserMemory {
	return getReturned(
		database
			.insert(userMemory)
			.values(newMemory)
			.onConflictDoUpdate({
				target: userMemory.id,
				set: {
					value: newMemory.value,
					confidence: newMemory.confidence,
					source: newMemory.source,
					writeCount: sql`write_count + 1`,
					lastWrite: newMemory.lastWrite,
					updatedAt: newMemory.updatedAt,
				},
			})
			.returning(),
	)
}

export function incrementBrainUserMemoryRead(
	database: BrainDatabase,
	id: string,
	timestamp: number,
): void {
	database
		.update(userMemory)
		.set({
			readCount: sql`read_count + 1`,
			lastRead: timestamp,
		})
		.where(eq(userMemory.id, id))
		.run()
}
```

### 4. [NEW] [write-user-memory.tool.ts](file:///Users/kibrukuture/Desktop/brioela/backend/src/agents/brain/_tools/write-user-memory.tool.ts)
Tool definition for `write_user_memory` utilizing standard Zod namespaces and merge validation.

```typescript
import { tool } from 'ai'
import { z } from '@brioela/shared/zod'
import type { BrainDatabase } from '@/agents/brain/_database'
import { 
	countBrainUserMemoryNamespaces, 
	readBrainUserMemory, 
	writeBrainUserMemory 
} from '@/agents/brain/_repositories'

const writeUserMemorySchema = z.object({
	namespace: z.string()
		.regex(/^[a-z][a-z0-9]*(\.[a-z][a-z0-9]*){0,2}$/, 'namespace must be dot-separated lowercase, max 3 levels')
		.describe('dot-separated namespace: e.g. health.medications or cooking.preferences'),
	key: z.string()
		.regex(/^[a-z][a-z0-9_]*$/, 'key must be lowercase with underscores only')
		.describe('the fact key within this namespace'),
	value: z.record(z.string(), z.json()).describe('the fact value to store — always a JSON object'),
	confidence: z.number().min(0).max(1).describe('0.0–1.0 confidence in this fact'),
	source: z.enum(['observed', 'stated', 'inferred']).describe('how this fact was determined'),
})

export const writeUserMemoryTool = (db: BrainDatabase, userId: string) => tool({
	description: 'Write or merge a structured fact into user_memory. Use namespace:key addressing. Merges intelligently — do not write the same fact twice.',
	parameters: writeUserMemorySchema,
	execute: async ({ namespace, key, value, confidence, source }) => {
		const id = `${namespace}:${key}`
		const existing = readBrainUserMemory(db, userId, namespace, key)

		// Check 40-namespace cap
		const isNewNamespace = !existing
		if (isNewNamespace) {
			const namespaceCount = countBrainUserMemoryNamespaces(db, userId)
			if (namespaceCount >= 40) {
				return { 
					error: 'namespace_cap_reached', 
					current: namespaceCount, 
					max: 40,
					reason: '40-namespace cap reached. Archive or consolidate before adding new namespaces.'
				}
			}
		}

		// Intelligent merge rules
		let mergedValue = value
		let mergedConfidence = confidence

		if (existing) {
			// Rule: Only overwrite if incoming confidence is higher, or source is 'stated'
			if (confidence <= existing.confidence && source !== 'stated') {
				return { 
					action: 'skipped', 
					reason: 'existing confidence is equal or higher and source is not stated' 
				}
			}

			try {
				const oldObj = JSON.parse(existing.value)
				mergedValue = { ...oldObj, ...value }
			} catch (e) {
				// Fallback to incoming value if old value was corrupted or not an object
				mergedValue = value
			}
		}

		const now = Date.now()
		const written = writeBrainUserMemory(db, {
			id,
			userId,
			namespace,
			key,
			value: JSON.stringify(mergedValue),
			confidence: mergedConfidence,
			source,
			isActive: true,
			writeCount: (existing?.writeCount ?? 0) + 1,
			lastWrite: now,
			updatedAt: now,
			readCount: existing?.readCount ?? 0,
			lastRead: existing?.lastRead ?? null,
			importance: existing?.importance ?? 5,
		})

		return {
			id: written.id,
			merged: !!existing,
			write_count: written.writeCount,
			status: 'written',
		}
	},
})
```

### 5. [NEW] [read-user-memory.tool.ts](file:///Users/kibrukuture/Desktop/brioela/backend/src/agents/brain/_tools/read-user-memory.tool.ts)
Tool definition for `read_user_memory`.

```typescript
import { tool } from 'ai'
import { z } from '@brioela/shared/zod'
import type { BrainDatabase } from '@/agents/brain/_database'
import { 
	incrementBrainUserMemoryRead, 
	listBrainUserMemories, 
	readBrainUserMemory 
} from '@/agents/brain/_repositories'

const readUserMemorySchema = z.object({
	namespace: z.string()
		.regex(/^[a-z][a-z0-9]*(\.[a-z][a-z0-9]*){0,2}$/, 'namespace must be dot-separated lowercase, max 3 levels')
		.describe('Which namespace to read from.'),
	key: z.string()
		.regex(/^[a-z][a-z0-9_]*$/, 'key must be lowercase with underscores only')
		.optional()
		.describe('Which key to read within the namespace. If omitted, returns all namespace keys.'),
})

export const readUserMemoryTool = (
	db: BrainDatabase, 
	userId: string,
	waitUntil?: (promise: Promise<any>) => void
) => tool({
	description: 'Read a specific fact or a full namespace from user_memory mid-session. Use only when context is not already loaded in prompt.',
	parameters: readUserMemorySchema,
	execute: async ({ namespace, key }) => {
		const now = Date.now()

		if (key) {
			const entry = readBrainUserMemory(db, userId, namespace, key)
			if (!entry) {
				return { found: false, id: `${namespace}:${key}` }
			}

			// Fire-and-forget stats update
			const updatePromise = Promise.resolve().then(() => {
				incrementBrainUserMemoryRead(db, entry.id, now)
			})
			if (waitUntil) {
				waitUntil(updatePromise)
			}

			try {
				return {
					found: true,
					id: entry.id,
					namespace: entry.namespace,
					key: entry.key,
					value: JSON.parse(entry.value),
					confidence: entry.confidence,
					last_write: entry.lastWrite,
				}
			} catch (e) {
				return {
					found: true,
					id: entry.id,
					namespace: entry.namespace,
					key: entry.key,
					value: entry.value,
					confidence: entry.confidence,
					last_write: entry.lastWrite,
				}
			}
		} else {
			const entries = listBrainUserMemories(db, userId, namespace)
			const parsedEntries = entries.map((entry) => {
				let parsedValue = entry.value
				try {
					parsedValue = JSON.parse(entry.value)
				} catch (_) {}

				return {
					key: entry.key,
					value: parsedValue,
					confidence: entry.confidence,
				}
			})

			return {
				namespace,
				count: parsedEntries.length,
				entries: parsedEntries,
			}
		}
	},
})
```

### 6. [NEW] [log-memory-event.tool.ts](file:///Users/kibrukuture/Desktop/brioela/backend/src/agents/brain/_tools/log-memory-event.tool.ts)
Tool definition for `log_memory_event`.

```typescript
import { tool } from 'ai'
import { z } from '@brioela/shared/zod'
import { nanoid } from 'nanoid'
import type { BrainDatabase } from '@/agents/brain/_database'
import { writeBrainMemoryEvent } from '@/agents/brain/_repositories'

const logMemoryEventSchema = z.object({
	kind: z.string().min(1).describe('What kind of event this is: e.g. food_intake, symptom_reported.'),
	payload: z.record(z.string(), z.json()).describe('The event data. Structure is free.'),
	captured_at: z.number().int().positive().optional().describe('Unix timestamp in ms when it occurred.'),
	source: z.string().min(1).describe('Who is logging this event: e.g. agent.'),
	entity_kind: z.string().optional().describe('What category of thing this is about: e.g. food.'),
	entity_id: z.string().optional().describe('The specific entity: e.g. recipe-id.'),
	geo_hash: z.string().optional().describe('Geohash of where the event occurred.'),
})

export const logMemoryEventTool = (
	db: BrainDatabase, 
	userId: string,
	activeSessionId: string | null = null,
) => tool({
	description: 'Log a raw real-world event into the memory event log.',
	parameters: logMemoryEventSchema,
	execute: async (eventParams) => {
		const id = nanoid(24)
		const now = Date.now()

		writeBrainMemoryEvent(db, {
			id,
			userId,
			kind: eventParams.kind,
			payloadJson: JSON.stringify(eventParams.payload),
			capturedAt: eventParams.captured_at ?? now,
			ingestedAt: now,
			source: eventParams.source,
			sessionId: activeSessionId,
			entityKind: eventParams.entity_kind ?? null,
			entityId: eventParams.entity_id ?? null,
			geoHash: eventParams.geo_hash ?? null,
		})

		return {
			id,
			status: 'logged',
		}
	},
})
```

### 7. [NEW] [index.ts](file:///Users/kibrukuture/Desktop/brioela/backend/src/agents/brain/_tools/index.ts)
Registration module matching the Session type permissions outlined in the build-guide.

```typescript
import { writeUserMemoryTool } from './write-user-memory.tool'
import { readUserMemoryTool } from './read-user-memory.tool'
import { logMemoryEventTool } from './log-memory-event.tool'
import type { BrainDatabase } from '@/agents/brain/_database'

// Define allowed caller session types
export type SessionCallerType = 'chat' | 'cooking' | 'alarm' | 'brain_maintenance' | 'behavior_pattern_detection'

const TOOL_PERMISSIONS: Record<SessionCallerType, string[]> = {
	chat: [
		'log_memory_event',
		'write_user_memory',
		'read_user_memory',
	],
	cooking: [
		'log_memory_event',
		'write_user_memory',
		'read_user_memory',
	],
	alarm: [
		'log_memory_event',
		'write_user_memory',
	],
	brain_maintenance: [
		'write_user_memory',
	],
	behavior_pattern_detection: [
		'log_memory_event',
		'write_user_memory',
	],
}

export function getBrainTools(
	db: BrainDatabase,
	userId: string,
	kind: SessionKind,
	activeSessionId: string | null = null,
	waitUntil?: (promise: Promise<void>) => void,
) {
	const allowed = new Set(TOOL_PERMISSIONS[kind])
	
	const all = {
		log_memory_event: logMemoryEventTool(db, userId, activeSessionId),
		write_user_memory: writeUserMemoryTool(db, userId),
		read_user_memory: readUserMemoryTool(db, userId, waitUntil),
	}

	return Object.fromEntries(
		Object.entries(all).filter(([name]) => allowed.has(name)),
	)
}
```
