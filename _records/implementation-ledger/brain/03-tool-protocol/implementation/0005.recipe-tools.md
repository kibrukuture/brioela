# 0005 — Recipe Tools

## Purpose

Implement the three AI-callable recipe tools: `view_user_recipe`, `update_user_recipe`, `archive_user_recipe`.

Recipes are structured cooking knowledge — not a general skill, but a specialized record type with dedicated fields (`title`, `ingredients`, `steps`, `servings`, `cuisine`, `tags`, `source`, `notes`). The recipe index (title + description only) is injected into cooking session prompts. Full content is loaded on demand via `view_user_recipe`.

There is no `create_user_recipe` tool — recipes are created by the MiraSession DO at the end of cooking sessions via a direct DB write. The agent can only view, update (refine after cooking), and archive (retire) recipes.

`update_user_recipe` uses the same version archive pattern as skills — writes a snapshot to `recipe_versions` before overwriting. `archive_user_recipe` soft-deletes by setting `status = 'archived'`.

import { and, eq } from '@brioela/shared/drizzle'
import { recipes } from '@/agents/brain/_schemas'
import type { BrainDatabase } from '@/agents/brain/_database'

export function readUserRecipe(db: BrainDatabase, id: string) {
	return db.select().from(recipes).where(eq(recipes.id, id)).get()
}

export function readActiveUserRecipe(db: BrainDatabase, id: string) {
	return db.select()
		.from(recipes)
		.where(and(eq(recipes.id, id), eq(recipes.status, 'active')))
		.get()
}

import { eq, sql } from '@brioela/shared/drizzle'
import { recipes, recipeVersions } from '@/agents/brain/_schemas'
import type { BrainDatabase } from '@/agents/brain/_database'

export function updateUserRecipe(
	db: BrainDatabase,
	id: string,
	patch: Partial<typeof recipes.$inferInsert>,
) {
	return db.update(recipes).set(patch).where(eq(recipes.id, id)).returning().get()
}

export function archiveRecipeVersion(db: BrainDatabase, values: typeof recipeVersions.$inferInsert) {
	return db.insert(recipeVersions).values(values).returning().get()
}

export function incrementRecipeCookCount(db: BrainDatabase, id: string, now: number) {
	return db.update(recipes)
		.set({ cookCount: sql`cook_count + 1`, lastCookedAt: now, updatedAt: now })
		.where(eq(recipes.id, id))
		.run()
}

### 3. [NEW] [view.user.recipe.tool.ts](file:///Users/kibrukuture/Desktop/brioela/backend/src/agents/brain/_tools/view.user.recipe.tool.ts)

```typescript
import { tool } from 'ai'
import { z } from '@brioela/shared/zod'
import type { BrainDatabase } from '@/agents/brain/_database'
import { readActiveUserRecipe, incrementRecipeCookCount } from '@/agents/brain/_repositories'
import { readCurrentEpochMs } from '@/time/_helpers'

const viewUserRecipeSchema = z.object({
	id: z.string().describe('Recipe id from the session recipe index.'),
})

export const viewUserRecipeTool = (db: BrainDatabase, waitUntil?: (p: Promise<void>) => void) => tool({
	description: 'Load the full content of a recipe into context on demand. Call only when the user asks about this specific recipe — not just to check if it exists.',
	inputSchema: viewUserRecipeSchema,
	execute: async (input) => {
		const recipe = readActiveUserRecipe(db, input.id)
		if (!recipe) return { found: false, id: input.id, hint: 'Recipe not found or archived.' }
		const now = readCurrentEpochMs()
		const update = Promise.resolve().then(() => { incrementRecipeCookCount(db, input.id, now) })
		if (waitUntil) waitUntil(update)
		
		// Content is a JSON string containing the detailed recipe properties
		const normalized = JSON.parse(recipe.content)
		return {
			id: recipe.id,
			title: recipe.title,
			source: recipe.source,
			source_session_id: recipe.sourceSessionId,
			source_url: recipe.sourceUrl,
			cook_count: recipe.cookCount,
			last_cooked_at: recipe.lastCookedAt,
			status: recipe.status,
			confidence: recipe.confidence,
			created_at: recipe.createdAt,
			updated_at: recipe.updatedAt,
			recipe_details: normalized,
		}
	},
})
```

### 4. [NEW] [update.user.recipe.tool.ts](file:///Users/kibrukuture/Desktop/brioela/backend/src/agents/brain/_tools/update.user.recipe.tool.ts)

```typescript
import { tool } from 'ai'
import { z } from '@brioela/shared/zod'
import { createId } from '@brioela/shared/_ids'
import type { BrainDatabase } from '@/agents/brain/_database'
import { readUserRecipe, updateUserRecipe, archiveRecipeVersion } from '@/agents/brain/_repositories'
import { readCurrentEpochMs } from '@/time/_helpers'

const updateUserRecipeSchema = z.object({
	id: z.string().describe('Recipe id.'),
	content: z.string().min(1).describe('Updated JSON string representing NormalizedImportedRecipe.'),
	reason: z.string().min(1).describe('Why the recipe is being updated — e.g. adjusted ratios, new technique learned.'),
	updated_by: z.enum(['agent', 'brain_maintenance']),
})

export const updateUserRecipeTool = (db: BrainDatabase, userId: string) => tool({
	description: 'Refine an existing recipe after a cooking session. Archives the current version to recipe_versions before overwriting. Use only to reflect real post-session learnings.',
	inputSchema: updateUserRecipeSchema,
	execute: async (input) => {
		const recipe = readUserRecipe(db, input.id)
		if (!recipe) return { error: 'recipe_not_found', id: input.id }
		if (recipe.status === 'archived') return { error: 'recipe_archived', id: input.id, hint: 'Restore the recipe before updating.' }
		const now = readCurrentEpochMs()
		
		db.transaction(() => {
			archiveRecipeVersion(db, {
				id: createId(),
				recipeId: recipe.id,
				userId,
				version: recipe.version,
				content: recipe.content,
				updatedBy: input.updated_by,
				updateReason: input.reason,
				archivedAt: now,
			})
			updateUserRecipe(db, input.id, {
				content: input.content,
				version: recipe.version + 1,
				updatedAt: now,
			})
		})
		return { id: input.id, previous_version: recipe.version, new_version: recipe.version + 1, status: 'updated' }
	},
})
```

### 5. [NEW] [archive.user.recipe.tool.ts](file:///Users/kibrukuture/Desktop/brioela/backend/src/agents/brain/_tools/archive.user.recipe.tool.ts)

```typescript
import { tool } from 'ai'
import { z } from '@brioela/shared/zod'
import type { BrainDatabase } from '@/agents/brain/_database'
import { readUserRecipe, updateUserRecipe } from '@/agents/brain/_repositories'
import { readCurrentEpochMs } from '@/time/_helpers'

const archiveUserRecipeSchema = z.object({
	id: z.string().describe('Recipe id.'),
	reason: z.string().min(1).describe('Why this recipe is being archived.'),
})

export const archiveUserRecipeTool = (db: BrainDatabase) => tool({
	description: 'Archive a recipe — removes it from the cooking session index. Row and content preserved. Use when the recipe is no longer relevant.',
	inputSchema: archiveUserRecipeSchema,
	execute: async (input) => {
		const recipe = readUserRecipe(db, input.id)
		if (!recipe) return { error: 'recipe_not_found', id: input.id }
		if (recipe.status === 'archived') return { error: 'already_archived', id: input.id }
		updateUserRecipe(db, input.id, { status: 'archived', updatedAt: readCurrentEpochMs() })
		return { id: input.id, title: recipe.title, status: 'archived', reason: input.reason }
	},
})
```
