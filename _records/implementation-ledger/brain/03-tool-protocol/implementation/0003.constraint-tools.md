# 0003 — Constraint Tools

## Purpose

Implement the two AI-callable constraint tools: `propose_user_constraint` and `confirm_user_constraint`.

Constraints are hard behavioral limits the agent must never violate. Every active constraint is injected into every session prompt in Block 4 (constraints block). The propose + confirm two-step is intentional — `propose_user_constraint` surfaces the candidate to the user; `confirm_user_constraint` is only called after the user explicitly says "yes". The agent must never auto-confirm.

Constraints have a `scope` field — `cooking`, `general`, or `global`. `global` applies to all session types. `cooking` only during Mira sessions. `general` only during Brain general sessions.

Constraints are never deleted — they are `revoked`. A revoked constraint stays in the DB with `status = 'revoked'` and is excluded from prompt injection. Revocation also requires explicit user confirmation before the agent calls `confirm_user_constraint` on the revoke path.

### 1. [NEW] [read.user.constraint.repository.ts](file:///Users/kibrukuture/Desktop/brioela/backend/src/agents/brain/_repositories/read.user.constraint.repository.ts)

```typescript
import { and, eq } from '@brioela/shared/drizzle'
import { constraints } from '@/agents/brain/_schemas'
import type { BrainDatabase } from '@/agents/brain/_database'

export function readUserConstraint(db: BrainDatabase, id: string) {
	return db.select().from(constraints).where(eq(constraints.id, id)).get()
}

export function listActiveUserConstraints(db: BrainDatabase, userId: string) {
	return db.select().from(constraints).where(and(eq(constraints.userId, userId), eq(constraints.status, 'active'))).all()
}
```

import { eq } from '@brioela/shared/drizzle'
import { constraints } from '@/agents/brain/_schemas'

export function insertUserConstraint(db: BrainDatabase, values: typeof constraints.$inferInsert) {
	return db.insert(constraints).values(values).returning().get()
}

export function updateUserConstraintStatus(
	db: BrainDatabase,
	id: string,
	patch: Partial<typeof constraints.$inferInsert>,
) {
	return db.update(constraints).set(patch).where(eq(constraints.id, id)).returning().get()
}

### 3. [NEW] [propose.user.constraint.tool.ts](file:///Users/kibrukuture/Desktop/brioela/backend/src/agents/brain/_tools/propose.user.constraint.tool.ts)

```typescript
import { tool } from 'ai'
import { z } from '@brioela/shared/zod'
import { createId } from '@brioela/shared/_ids'
import type { BrainDatabase } from '@/agents/brain/_database'
import { insertUserConstraint } from '@/agents/brain/_repositories'
import { readCurrentEpochMs } from '@/time/_helpers'

const proposeUserConstraintSchema = z.object({
	text: z.string().min(1).max(400).describe('The constraint statement as it will appear in the system prompt. Must be actionable and clear.'),
	scope: z.enum(['global', 'general', 'cooking']).describe('Which session types this constraint applies to.'),
	reason: z.string().min(1).describe('Why this constraint is being proposed. Will be shown to the user.'),
})

export const proposeUserConstraintTool = (db: BrainDatabase, userId: string) => tool({
	description: 'Propose a new behavioral constraint. Creates a pending row — must be confirmed by the user before it becomes active. Never auto-confirm.',
	inputSchema: proposeUserConstraintSchema,
	execute: async (input) => {
		const now = readCurrentEpochMs()
		const constraint = insertUserConstraint(db, {
			id: createId(),
			userId,
			text: input.text,
			scope: input.scope,
			status: 'pending',
			reason: input.reason,
			revokedReason: null,
			revokedAt: null,
			createdAt: now,
			updatedAt: now,
		})
		return {
			id: constraint.id,
			status: 'pending',
			text: constraint.text,
			scope: constraint.scope,
			user_action_required: 'Ask the user to confirm this constraint before calling confirm_user_constraint.',
		}
	},
})
```

### 4. [NEW] [confirm.user.constraint.tool.ts](file:///Users/kibrukuture/Desktop/brioela/backend/src/agents/brain/_tools/confirm.user.constraint.tool.ts)

```typescript
import { tool } from 'ai'
import { z } from '@brioela/shared/zod'
import type { BrainDatabase } from '@/agents/brain/_database'
import { readUserConstraint, updateUserConstraintStatus } from '@/agents/brain/_repositories'
import { readCurrentEpochMs } from '@/time/_helpers'

const confirmUserConstraintSchema = z.object({
	id: z.string().describe('Constraint id from propose_user_constraint response.'),
	action: z.enum(['activate', 'revoke']).describe('activate = make pending constraint active. revoke = revoke an active constraint.'),
	revoke_reason: z.string().optional().describe('Required when action = revoke. Explains why being revoked.'),
})

export const confirmUserConstraintTool = (db: BrainDatabase) => tool({
	description: 'Confirm a pending constraint as active, or revoke an active one. Only call after explicit user confirmation — never speculatively.',
	inputSchema: confirmUserConstraintSchema,
	execute: async (input) => {
		const constraint = readUserConstraint(db, input.id)
		if (!constraint) return { error: 'constraint_not_found', id: input.id }

		const now = readCurrentEpochMs()
		if (input.action === 'activate') {
			if (constraint.status !== 'pending') return { error: 'not_pending', id: input.id, current_status: constraint.status }
			updateUserConstraintStatus(db, input.id, { status: 'active', updatedAt: now })
			return { id: input.id, status: 'active', text: constraint.text, scope: constraint.scope }
		}

		if (input.action === 'revoke') {
			if (constraint.status !== 'active') return { error: 'not_active', id: input.id, current_status: constraint.status }
			if (!input.revoke_reason) return { error: 'revoke_reason_required' }
			updateUserConstraintStatus(db, input.id, { status: 'revoked', revokedReason: input.revoke_reason, revokedAt: now, updatedAt: now })
			return { id: input.id, status: 'revoked', revoke_reason: input.revoke_reason }
		}

		return { error: 'invalid_action', action: input.action }
	},
})
```
