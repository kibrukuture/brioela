# 0006 — Session Tools

## Purpose

Implement the two AI-callable session tools: `load_session_context` and `search_session_history`.

`load_session_context` retrieves the compressed summary of a prior session by ID. It is called when the agent needs to recall a specific past session beyond the current context window. Returns the `outcomeSummary` blob — the 4-field Haiku-compressed record written by `compress.session.handler.ts` (Scope 05).

`search_session_history` performs full-text search over `session_turns` using the FTS5 index created in `02-sqlite-migration-runtime/implementation/0002.fts5-and-sync-triggers.md`. It returns a ranked list of turn excerpts with their parent session IDs and timestamps. Used when the agent needs to locate what was said or decided in the past without knowing which session it belongs to.

Both tools are read-only. No writes. Both are available only in `general` sessions — alarm and cooking sessions do not get history access.

### 1. [NEW] [read.session.repository.ts](file:///Users/kibrukuture/Desktop/brioela/backend/src/agents/brain/_repositories/read.session.repository.ts)

```typescript
import { and, eq, desc } from '@brioela/shared/drizzle'
import { sessions, sessionTurns } from '@/agents/brain/_schemas'
import type { BrainDatabase } from '@/agents/brain/_database'

export function readSessionById(db: BrainDatabase, sessionId: string, userId: string) {
	return db.select()
		.from(sessions)
		.where(and(eq(sessions.id, sessionId), eq(sessions.userId, userId)))
		.get()
}

export function searchSessionTurnsFts(db: BrainDatabase, userId: string, query: string, limit: number) {
	// Raw SQL for FTS5 MATCH — Drizzle does not abstract FTS5 queries
	return db.all<{
		session_id: string
		turn_index: number
		role: string
		content_excerpt: string
		created_at: number
		rank: number
	}>(
		`
		SELECT
			st.session_id,
			st.turn_index,
			st.role,
			snippet(session_turns_fts, 0, '<b>', '</b>', '...', 32) AS content_excerpt,
			st.created_at,
			bm25(session_turns_fts) AS rank
		FROM session_turns_fts
		JOIN session_turns st ON session_turns_fts.rowid = st.rowid
		JOIN sessions s ON st.session_id = s.id
		WHERE session_turns_fts MATCH ?
		AND s.user_id = ?
		ORDER BY rank
		LIMIT ?
		`,
		[query, userId, limit],
	)
}
```

### 2. [NEW] [load.session.context.tool.ts](file:///Users/kibrukuture/Desktop/brioela/backend/src/agents/brain/_tools/load.session.context.tool.ts)

```typescript
import { tool } from 'ai'
import { z } from '@brioela/shared/zod'
import type { BrainDatabase } from '@/agents/brain/_database'
import { readSessionById } from '@/agents/brain/_repositories'

const loadSessionContextSchema = z.object({
	session_id: z.string().describe('Session id to load compressed context for.'),
})

export const loadSessionContextTool = (db: BrainDatabase, userId: string) => tool({
	description: 'Load the compressed outcome summary of a prior session. Use when recalling a specific past session by ID.',
	inputSchema: loadSessionContextSchema,
	execute: async (input) => {
		const session = readSessionById(db, input.session_id, userId)
		if (!session) return { found: false, session_id: input.session_id }
		return {
			session_id: session.id,
			session_type: session.sessionType,
			status: session.status,
			started_at: session.startedAt,
			ended_at: session.endedAt,
			outcome_summary: session.outcomeSummary ?? null,
			turn_count: session.turnCount,
			was_compressed: session.wasCompressed,
		}
	},
})
```

### 3. [NEW] [search.session.history.tool.ts](file:///Users/kibrukuture/Desktop/brioela/backend/src/agents/brain/_tools/search.session.history.tool.ts)

```typescript
import { tool } from 'ai'
import { z } from '@brioela/shared/zod'
import type { BrainDatabase } from '@/agents/brain/_database'
import { searchSessionTurnsFts } from '@/agents/brain/_repositories'

const searchSessionHistorySchema = z.object({
	query: z.string().min(1).max(200).describe('FTS5 search query. Standard FTS5 MATCH syntax. Keep it simple — 2-5 key terms.'),
	limit: z.number().int().min(1).max(20).default(10).describe('Max results to return. Default 10, max 20.'),
})

export const searchSessionHistoryTool = (db: BrainDatabase, userId: string) => tool({
	description: 'Full-text search over past session turns. Returns ranked excerpts with session IDs. Use to locate past decisions or discussions when the session ID is unknown.',
	inputSchema: searchSessionHistorySchema,
	execute: async (input) => {
		const results = searchSessionTurnsFts(db, userId, input.query, input.limit)
		return {
			query: input.query,
			count: results.length,
			results: results.map(r => ({
				session_id: r.session_id,
				turn_index: r.turn_index,
				role: r.role,
				excerpt: r.content_excerpt,
				created_at: r.created_at,
			})),
		}
	},
})
```
