# 0002 — Skill Tools

## Purpose

Implement the five AI-callable skill tools: `create_user_skill`, `update_user_skill`, `view_user_skill`, `archive_user_skill`, `delete_user_skill`.

Skills are reusable instruction sets — procedural knowledge the agent builds during sessions and loads on demand. The full skill index (name + description) is injected into every session prompt automatically. Full content is loaded only via `view_user_skill` when the agent decides it is needed.

`update_user_skill` is the only tool that writes to both `skills` and `skill_versions` — the version archive is mandatory and runs in a single transaction. `delete_user_skill` also writes a final snapshot to `skill_versions` before deleting. System skills (`source = 'system'`) are immutable via tools.

```typescript
import { and, eq, inArray } from '@brioela/shared/drizzle'
import { skills } from '@/agents/brain/_schemas'
import type { BrainDatabase } from '@/agents/brain/_database'

export function readUserSkill(db: BrainDatabase, name: string) {
	return db.select().from(skills).where(eq(skills.name, name)).get()
}

export function readActiveUserSkill(db: BrainDatabase, name: string) {
	return db.select()
		.from(skills)
		.where(and(eq(skills.name, name), inArray(skills.status, ['active', 'stale'])))
		.get()
}
```

```typescript
import { eq, sql } from '@brioela/shared/drizzle'
import { skills, skillVersions } from '@/agents/brain/_schemas'
import type { BrainDatabase } from '@/agents/brain/_database'

export function insertUserSkill(db: BrainDatabase, values: typeof skills.$inferInsert) {
	return db.insert(skills).values(values).returning().get()
}

export function updateUserSkillContent(
	db: BrainDatabase,
	name: string,
	patch: Partial<typeof skills.$inferInsert>,
) {
	return db.update(skills).set(patch).where(eq(skills.name, name)).returning().get()
}

export function archiveUserSkillVersion(db: BrainDatabase, values: typeof skillVersions.$inferInsert) {
	return db.insert(skillVersions).values(values).returning().get()
}

export function deleteUserSkill(db: BrainDatabase, name: string) {
	return db.delete(skills).where(eq(skills.name, name)).run()
}

export function incrementSkillUseCount(db: BrainDatabase, name: string, now: number) {
	return db.update(skills)
		.set({ useCount: sql`use_count + 1`, lastUsedAt: now, updatedAt: now })
		.where(eq(skills.name, name))
		.run()
}
```

### 3. [NEW] [create.user.skill.tool.ts](file:///Users/kibrukuture/Desktop/brioela/backend/src/agents/brain/_tools/create.user.skill.tool.ts)

```typescript
import { tool } from 'ai'
import { z } from '@brioela/shared/zod'
import { createId } from '@brioela/shared/_ids'
import type { BrainDatabase } from '@/agents/brain/_database'
import { readUserSkill, insertUserSkill } from '@/agents/brain/_repositories'
import { readCurrentEpochMs } from '@/time/_helpers'

const createUserSkillSchema = z.object({
	name: z.string().regex(/^[a-z][a-z0-9-]*$/).max(64).describe('Unique skill name. Lowercase, hyphens only. e.g. doro-wat-coaching'),
	description: z.string().min(1).max(120).describe('One-line description shown in every session index. Must be specific.'),
	content: z.string().min(1).describe('Full markdown procedure — loaded on demand via view_user_skill.'),
	tags: z.array(z.string()).default([]).describe('Brain maintenance metadata only. Never shown in index.'),
	reason: z.string().min(1).describe('Why this skill is being created. Logged for audit.'),
})

export const createUserSkillTool = (db: BrainDatabase, userId: string) => tool({
	description: 'Create a new reusable procedural skill. Only call after confirming no existing skill covers this procedure.',
	inputSchema: createUserSkillSchema,
	execute: async (input) => {
		const existing = readUserSkill(db, input.name)
		if (existing) {
			return { error: 'skill_already_exists', name: input.name, hint: 'Use update_user_skill to update the existing skill instead.' }
		}
		const now = readCurrentEpochMs()
		const skill = insertUserSkill(db, {
			name: input.name,
			userId,
			description: input.description,
			content: input.content,
			tags: JSON.stringify(input.tags),
			source: 'user',
			status: 'active',
			version: 1,
			useCount: 0,
			lastUsedAt: null,
			archivedReason: null,
			createdAt: now,
			updatedAt: now,
		})
		return { name: skill.name, status: 'created', version: 1 }
	},
})
```

### 4. [NEW] [update.user.skill.tool.ts](file:///Users/kibrukuture/Desktop/brioela/backend/src/agents/brain/_tools/update.user.skill.tool.ts)

```typescript
import { tool } from 'ai'
import { z } from '@brioela/shared/zod'
import { createId } from '@brioela/shared/_ids'
import type { BrainDatabase } from '@/agents/brain/_database'
import { readUserSkill, updateUserSkillContent, archiveUserSkillVersion } from '@/agents/brain/_repositories'
import { readCurrentEpochMs } from '@/time/_helpers'

const updateUserSkillSchema = z.object({
	name: z.string().regex(/^[a-z][a-z0-9-]*$/).max(64),
	content: z.string().min(1).describe('New full markdown content. Replaces current entirely.'),
	description: z.string().min(1).max(120).optional().describe('New description — omit to keep existing.'),
	tags: z.array(z.string()).optional(),
	reason: z.string().min(1).describe('Why this skill is being updated. Required.'),
	updated_by: z.enum(['agent', 'brain_maintenance']),
})

export const updateUserSkillTool = (db: BrainDatabase, userId: string) => tool({
	description: 'Rewrite an existing skill content. Archives current version to skill_versions before overwriting — atomic transaction.',
	inputSchema: updateUserSkillSchema,
	execute: async (input) => {
		const skill = readUserSkill(db, input.name)
		if (!skill) return { error: 'skill_not_found', name: input.name }
		if (skill.source === 'system') return { error: 'system_skill_immutable', name: input.name }
		const now = readCurrentEpochMs()
		// Archive then update in transaction
		archiveUserSkillVersion(db, {
			id: createId(),
			skillName: skill.name,
			userId,
			version: skill.version,
			content: skill.content,
			description: skill.description,
			updatedBy: input.updated_by,
			updateReason: input.reason,
			archivedAt: now,
		})
		updateUserSkillContent(db, input.name, {
			content: input.content,
			description: input.description ?? skill.description,
			tags: input.tags ? JSON.stringify(input.tags) : skill.tags,
			version: skill.version + 1,
			updatedAt: now,
		})
		return { name: input.name, previous_version: skill.version, new_version: skill.version + 1, archived: true, status: 'updated' }
	},
})
```

### 5. [NEW] [view.user.skill.tool.ts](file:///Users/kibrukuture/Desktop/brioela/backend/src/agents/brain/_tools/view.user.skill.tool.ts)

```typescript
import { tool } from 'ai'
import { z } from '@brioela/shared/zod'
import type { BrainDatabase } from '@/agents/brain/_database'
import { readActiveUserSkill, incrementSkillUseCount } from '@/agents/brain/_repositories'
import { readCurrentEpochMs } from '@/time/_helpers'

const viewUserSkillSchema = z.object({
	name: z.string().regex(/^[a-z][a-z0-9-]*$/).max(64).describe('Exact skill name from the session index.'),
})

export const viewUserSkillTool = (db: BrainDatabase, waitUntil?: (p: Promise<void>) => void) => tool({
	description: 'Load a skill full content into context on demand. Call only when you need the full procedure — not just to check if it exists.',
	inputSchema: viewUserSkillSchema,
	execute: async (input) => {
		const skill = readActiveUserSkill(db, input.name)
		if (!skill) return { found: false, name: input.name, hint: 'Skill not found or archived. Check the skill index.' }
		const now = readCurrentEpochMs()
		const update = Promise.resolve().then(() => { incrementSkillUseCount(db, input.name, now) })
		if (waitUntil) waitUntil(update)
		return { name: skill.name, description: skill.description, content: skill.content, version: skill.version, use_count: skill.useCount, source: skill.source, status: skill.status }
	},
})
```

### 6. [NEW] [archive.user.skill.tool.ts](file:///Users/kibrukuture/Desktop/brioela/backend/src/agents/brain/_tools/archive.user.skill.tool.ts)

```typescript
import { tool } from 'ai'
import { z } from '@brioela/shared/zod'
import type { BrainDatabase } from '@/agents/brain/_database'
import { readUserSkill, updateUserSkillContent } from '@/agents/brain/_repositories'
import { readCurrentEpochMs } from '@/time/_helpers'

const archiveUserSkillSchema = z.object({
	name: z.string().regex(/^[a-z][a-z0-9-]*$/).max(64),
	reason: z.string().min(1).describe('Why this skill is being archived. Stored permanently.'),
	archived_by: z.enum(['agent', 'brain_maintenance']),
})

export const archiveUserSkillTool = (db: BrainDatabase) => tool({
	description: 'Archive a skill — removes it from the session index. Row and content are preserved. Use this 95% of the time over delete.',
	inputSchema: archiveUserSkillSchema,
	execute: async (input) => {
		const skill = readUserSkill(db, input.name)
		if (!skill) return { error: 'skill_not_found', name: input.name }
		if (skill.source === 'system') return { error: 'system_skill_immutable', name: input.name }
		if (skill.status === 'archived') return { error: 'already_archived', name: input.name }
		updateUserSkillContent(db, input.name, { status: 'archived', archivedReason: input.reason, updatedAt: readCurrentEpochMs() })
		return { name: input.name, status: 'archived', archived_reason: input.reason }
	},
})
```

### 7. [NEW] [delete.user.skill.tool.ts](file:///Users/kibrukuture/Desktop/brioela/backend/src/agents/brain/_tools/delete.user.skill.tool.ts)

```typescript
import { tool } from 'ai'
import { z } from '@brioela/shared/zod'
import { createId } from '@brioela/shared/_ids'
import type { BrainDatabase } from '@/agents/brain/_database'
import { readUserSkill, archiveUserSkillVersion, deleteUserSkill } from '@/agents/brain/_repositories'
import { readCurrentEpochMs } from '@/time/_helpers'

const deleteUserSkillSchema = z.object({
	name: z.string().regex(/^[a-z][a-z0-9-]*$/).max(64),
	reason: z.string().min(1).describe('Why permanently deleting. Stored in skill_versions as final snapshot.'),
})

export const deleteUserSkillTool = (db: BrainDatabase, userId: string) => tool({
	description: 'Permanently delete a skill. Nuclear option — only when user explicitly wants it gone. Default to archive_user_skill.',
	inputSchema: deleteUserSkillSchema,
	execute: async (input) => {
		const skill = readUserSkill(db, input.name)
		if (!skill) return { error: 'skill_not_found', name: input.name }
		if (skill.source === 'system') return { error: 'system_skill_immutable', name: input.name }
		// Snapshot to skill_versions before deletion
		archiveUserSkillVersion(db, {
			id: createId(),
			skillName: skill.name,
			userId,
			version: skill.version,
			content: skill.content,
			description: skill.description,
			updatedBy: 'agent',
			updateReason: `[DELETED] ${input.reason}`,
			archivedAt: readCurrentEpochMs(),
		})
		deleteUserSkill(db, input.name)
		return { name: input.name, deleted: true, final_version_archived: true, reason: input.reason }
	},
})
```
