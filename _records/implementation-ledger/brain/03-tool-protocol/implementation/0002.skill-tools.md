# 0002 — Skill Tools

## Purpose

Implement the five AI-callable skill tools: `create_user_skill`, `update_user_skill`, `view_user_skill`, `archive_user_skill`, `delete_user_skill`.

Skills are reusable instruction sets — procedural knowledge the agent builds during sessions and loads on demand. The full skill index (name + description) is injected into every session prompt automatically. Full content is loaded only via `view_user_skill` when the agent decides it is needed.

`update_user_skill` is the only tool that writes to both `skills` and `skill_versions` — the version archive is mandatory and runs in a single transaction. `delete_user_skill` also writes a final snapshot to `skill_versions` before deleting. System skills (`source = 'system'`) are immutable via tools.

---

## Implementation contract (complaints 007–010)

- **007:** Each tool splits across `_schemas/`, `_prompts/`, `_executables/`, and `.tool.ts` (see shipped recipe/alarm tools).
- **008:** Repos use `getOne(...)` / `getReturned(...)` from `@/database/drizzle/_database` — never `.returning().get()` or bare `.get()` on selects.
- **009:** Row IDs use `createId()` from `@brioela/shared/_ids`.
- **010:** Drizzle operators import from `@/database/drizzle/_database`; schema builders from `@/database/sqlite/_schema`.

### 1. [NEW] [read.user.skill.repository.ts](file:///Users/kibrukuture/Desktop/brioela/backend/src/agents/brain/_repositories/read.user.skill.repository.ts)

```typescript
import { and, eq, inArray, getOne } from '@/database/drizzle/_database'
import { skills } from '@/agents/brain/_schemas'
import type { BrainDatabase } from '@/agents/brain/_database'

export function readUserSkill(database: BrainDatabase, name: string) {
	return getOne(
		database
			.select()
			.from(skills)
			.where(eq(skills.name, name)),
	)
}

export function readActiveUserSkill(database: BrainDatabase, name: string) {
	return getOne(
		database
			.select()
			.from(skills)
			.where(and(eq(skills.name, name), inArray(skills.status, ['active', 'stale']))),
	)
}
```

### 2. [NEW] [write.user.skill.repository.ts](file:///Users/kibrukuture/Desktop/brioela/backend/src/agents/brain/_repositories/write.user.skill.repository.ts)

```typescript
import { eq, getReturned, sql } from '@/database/drizzle/_database'
import { skills, skillVersions } from '@/agents/brain/_schemas'
import type { BrainDatabase } from '@/agents/brain/_database'

export function insertUserSkill(database: BrainDatabase, values: typeof skills.$inferInsert) {
	return getReturned(
		database
			.insert(skills)
			.values(values)
			.returning(),
	)
}

export function updateUserSkillContent(
	database: BrainDatabase,
	name: string,
	patch: Partial<typeof skills.$inferInsert>,
) {
	return getReturned(
		database
			.update(skills)
			.set(patch)
			.where(eq(skills.name, name))
			.returning(),
	)
}

export function archiveUserSkillVersion(database: BrainDatabase, values: typeof skillVersions.$inferInsert) {
	return getReturned(
		database
			.insert(skillVersions)
			.values(values)
			.returning(),
	)
}

export function deleteUserSkill(database: BrainDatabase, name: string) {
	return database.delete(skills).where(eq(skills.name, name)).run()
}

export function incrementSkillUseCount(database: BrainDatabase, name: string, now: number) {
	return database.update(skills)
		.set({ useCount: sql`use_count + 1`, lastUsedAt: now, updatedAt: now })
		.where(eq(skills.name, name))
		.run()
}
```

### 3. Skill tools — split file layout (complaint 007)

Each skill tool ships as four files. Example for `create_user_skill`:

| File | Role |
|---|---|
| `_schemas/create.user.skill.schema.ts` | Zod input |
| `_prompts/create.user.skill.prompt.ts` | Tool description |
| `_executables/create.user.skill.executable.ts` | `createUserSkillExecutable` — uses `createId()` |
| `create.user.skill.tool.ts` | Thin AI SDK wrapper |

```typescript
// create.user.skill.tool.ts
import { tool } from 'ai'
import type { BrainDatabase } from '@/agents/brain/_database'
import { createUserSkillSchema } from '@/agents/brain/_tools/_schemas/create.user.skill.schema'
import { createUserSkillPrompt } from '@/agents/brain/_tools/_prompts/create.user.skill.prompt'
import { createUserSkillExecutable } from '@/agents/brain/_tools/_executables/create.user.skill.executable'

export const createUserSkillTool = (database: BrainDatabase, userId: string) => tool({
	description: createUserSkillPrompt,
	inputSchema: createUserSkillSchema,
	execute: async (params) => createUserSkillExecutable(database, userId, params),
})
```

Repeat the same split for `update_user_skill`, `view_user_skill`, `archive_user_skill`, and `delete_user_skill`. Reference shipped `view.user.recipe.tool.ts` + siblings.
